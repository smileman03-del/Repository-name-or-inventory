/** * 수술실 소모품 재고관리 - Google Apps Script * * 사용법: * 1. 이 코드를 Apps Script 편집기에 전체 붙여넣기 * 2. 저장 (Ctrl+S) * 3. 배포 → 새 배포 → 웹 앱 → 액세스 권한: 모든 사용자 → 배포 * 4. 생성된 URL을 앱 코드의 SCRIPT_URL에 붙여넣기 */ const SHEET_NAMES = { MASTER: '품목 마스터', WEEKLY: '주간 재고 파악', REALTIME: '실시간 재고 대장', REQUEST: '물품 청구서' }; // ==================================================== // GET: 데이터 조회 (현황, 청구서, 주간실사 불러오기) // ==================================================== function doGet(e) { const action = e.parameter.action || 'getAll'; let result = {}; try { if (action === 'getAll') { result = { master: getMasterData(), weekly: getWeeklyData() }; } else if (action === 'getMaster') { result = { master: getMasterData() }; } else if (action === 'getWeekly') { result = { weekly: getWeeklyData() }; } return jsonResponse({ status: 'success', data: result }); } catch (err) { return jsonResponse({ status: 'error', message: err.toString() }); } } // ==================================================== // POST: 데이터 입력/수정 (주간실사 입력, 청구 제출) // ==================================================== function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; let result = {}; if (action === 'updateWeeklyStock') { result = updateWeeklyStock(body.itemName, body.date, body.stock); } else if (action === 'generateWeeklyList') { result = generateWeeklyList(); } else if (action === 'submitRequest') { result = submitRequest(body.items); } else if (action === 'updateMasterStock') { result = updateMasterStock(body.itemName, body.stock); } else { throw new Error('알 수 없는 action: ' + action); } return jsonResponse({ status: 'success', result: result }); } catch (err) { return jsonResponse({ status: 'error', message: err.toString() }); } } // ==================================================== // 품목 마스터 데이터 가져오기 // ==================================================== function getMasterData() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.MASTER); const lastRow = sheet.getLastRow(); if (lastRow < 2) return []; const data = sheet.getRange(2, 1, lastRow - 1, 8).getValues(); return data .filter(row => row[1] !== '') .map(row => ({ num: row[0], name: row[1], category: row[2], loc: row[3], spec: row[4], minStock: Number(row[5]) || 0, curStock: Number(row[6]) || 0, status: row[7] })); } // ==================================================== // 주간 재고 파악 데이터 가져오기 // ==================================================== function getWeeklyData() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.WEEKLY); const lastRow = sheet.getLastRow(); if (lastRow < 2) return []; const data = sheet.getRange(2, 1, lastRow - 1, 6).getValues(); return data .filter(row => row[2] !== '') .map((row, idx) => ({ rowIndex: idx + 2, date: formatDate(row[0]), type: row[1], name: row[2], stock: row[3], spec: row[4], loc: row[5] })); } // ==================================================== // 주간실사: 오늘 날짜로 품목 마스터 전체 자동 입력 // ==================================================== function generateWeeklyList() { const masterSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.MASTER); const weeklySheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.WEEKLY); const lastRow = masterSheet.getLastRow(); const masterData = masterSheet.getRange(2, 1, lastRow - 1, 8).getValues() .filter(row => row[1] !== ''); const today = new Date(); const todayStr = formatDate(today); // 기존 오늘 날짜 데이터 확인 및 삭제 const weeklyLastRow = weeklySheet.getLastRow(); if (weeklyLastRow >= 2) { const weeklyData = weeklySheet.getRange(2, 1, weeklyLastRow - 1, 6).getValues(); const firstDate = weeklyData.length > 0 ? formatDate(weeklyData[0][0]) : ''; if (firstDate === todayStr) { weeklySheet.deleteRows(2, weeklyLastRow - 1); } } // 새 데이터 작성 const newRows = masterData.map(row => [ today, '주간실사', row[1], '', row[4], row[3] ]); if (newRows.length > 0) { weeklySheet.getRange(2, 1, newRows.length, 6).setValues(newRows); } return { count: newRows.length, date: todayStr }; } // ==================================================== // 주간실사: 개별 품목 재고량 입력/수정 // ==================================================== function updateWeeklyStock(itemName, date, stock) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.WEEKLY); const lastRow = sheet.getLastRow(); if (lastRow < 2) throw new Error('주간 재고 파악 시트가 비어있습니다'); const data = sheet.getRange(2, 1, lastRow - 1, 6).getValues(); for (let i = 0; i < data.length; i++) { const rowDate = formatDate(data[i][0]); if (data[i][2] === itemName && rowDate === date) { sheet.getRange(i + 2, 4).setValue(stock); // 품목 마스터의 현재재고도 함께 업데이트 updateMasterStock(itemName, stock); return { updated: true, row: i + 2 }; } } throw new Error('해당 품목을 찾을 수 없습니다: ' + itemName); } // ==================================================== // 품목 마스터: 현재재고 직접 수정 // ==================================================== function updateMasterStock(itemName, stock) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.MASTER); const lastRow = sheet.getLastRow(); const data = sheet.getRange(2, 1, lastRow - 1, 8).getValues(); for (let i = 0; i < data.length; i++) { if (data[i][1] === itemName) { sheet.getRange(i + 2, 7).setValue(stock); const minStock = Number(data[i][5]) || 0; const newStatus = Number(stock) < minStock ? '❌ 재고부족' : '✅ 안정'; sheet.getRange(i + 2, 8).setValue(newStatus); return { updated: true, row: i + 2, status: newStatus }; } } throw new Error('해당 품목을 찾을 수 없습니다: ' + itemName); } // ==================================================== // 물품 청구서 제출: 청구서 시트에 기록 // ==================================================== function submitRequest(items) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAMES.REQUEST); // 기존 데이터 초기화 (6행부터) const lastRow = sheet.getLastRow(); if (lastRow >= 6) { sheet.getRange(6, 1, lastRow - 5, 5).clearContent(); } // 청구일자 갱신 sheet.getRange('C2').setValue(new Date()); // 새 청구 항목 작성 const rows = items.map(item => [ item.name, item.spec, item.curStock, item.minStock, item.qty ]); if (rows.length > 0) { sheet.getRange(6, 1, rows.length, 5).setValues(rows); } return { count: rows.length, date: formatDate(new Date()) }; } // ==================================================== // 유틸: 날짜 포맷 (YYYY-MM-DD) // ==================================================== function formatDate(date) { if (!date) return ''; if (typeof date === 'string') return date; const tz = Session.getScriptTimeZone(); return Utilities.formatDate(new Date(date), tz, 'yyyy-MM-dd'); } // ==================================================== // 유틸: JSON 응답 + CORS 허용 // ==================================================== function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
